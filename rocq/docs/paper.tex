\section{Source Language $\lambda_{\mathsf{SCE}}$}
%%% ============================================================

\subsection{Syntax}

\[
\begin{array}{lrcl}
\text{Module Types}  & \mathcal{M} & ::= & \kw{Intf}\;\mathcal{A}
                                     \mid  \mathcal{A} \tarr_m \mathcal{M} \\[4pt]
\text{Source Types}   & \mathcal{A}, \mathcal{B} & ::= &
    \kw{Int} \mid \varepsilon
    \mid \mathcal{A} \tarr \mathcal{B}
    \mid \mathcal{A} \tand \mathcal{B}
    \mid \{l : \mathcal{A}\}
    \mid \kw{Mod}\;\mathcal{M} \\[4pt]
\text{Sandbox}   & \sigma & ::= & \smark \mid \omark \\[4pt]
\text{Source Exp.} & \shat{e}, \shat{v} & ::= &
    \squery \mid \hat{\varepsilon} \mid \hat{i}
    \mid \slam\,\mathcal{A}.\,\shat{e}
    \mid \shat{e}_1\;\shat{e}_2
    \mid \shat{e}_1 \sdmrg \shat{e}_2
    \mid \shat{e}_1 \snmrg \shat{e}_2 \\
  & & \mid & \shat{e}_1 \sbox \shat{e}_2
    \mid \sclos{\shat{v}}{\mathcal{A}}{\shat{e}}
    \mid \srec{l}{\shat{e}}
    \mid \shat{e}.l
    \mid \shat{e}.n \\
  & & \mid & \slet{\shat{e}_1}{\mathcal{A}}{\shat{e}_2}
    \mid \sopen{\shat{e}_1}{\shat{e}_2} \\
  & & \mid & \Struct{\sigma}{\shat{e}}
    \mid \Functor{\sigma}{\mathcal{A}}{\shat{e}} \\
  & & \mid & \mclos{\shat{v}}{\mathcal{A}}{\shat{e}}
    \mid \Link{\shat{e}_1}{\shat{e}_2}
\end{array}
\]

\subsection{Type Elaboration}

$\lsem \cdot \rsem$ maps source types to core types:
\[
\begin{array}{rcl@{\qquad\qquad}rcl}
  \lsem \kw{Int} \rsem &=& \kw{Int}
    & \lsem \{l:\mathcal{A}\} \rsem &=& \{l:\lsem \mathcal{A} \rsem\} \\
  \lsem \varepsilon \rsem &=& \varepsilon
    & \lsem \kw{Mod}\;(\kw{Intf}\;\mathcal{A}) \rsem &=& \lsem \mathcal{A} \rsem \\
  \lsem \mathcal{A} \tarr \mathcal{B} \rsem &=& \lsem \mathcal{A} \rsem \tarr \lsem \mathcal{B} \rsem
    & \lsem \kw{Mod}\;(\mathcal{A} \tarr_m \mathcal{M}) \rsem &=& \lsem \mathcal{A} \rsem \tarr \lsem \kw{Mod}\;\mathcal{M} \rsem \\
  \lsem \mathcal{A} \tand \mathcal{B} \rsem &=& \lsem \mathcal{A} \rsem \tand \lsem \mathcal{B} \rsem
\end{array}
\]

\subsection{Type-Directed Elaboration}
%%% ============================================================

Judgment: $\boxed{\mathcal{G} \vdash \shat{e} : \mathcal{A} \elmark e}$

\subsubsection*{Core Expression Rules}

\begin{mathpar}
\inferrule*[Right=E-Ctx]
  { }
  { \mathcal{G} \vdash \squery : \mathcal{G} \elmark \query }

\inferrule*[Right=E-Lit]
  { }
  { \mathcal{G} \vdash \hat{i} : \kw{Int} \elmark i }

\inferrule*[Right=E-Unit]
  { }
  { \mathcal{G} \vdash \hat{\varepsilon} : \varepsilon \elmark \varepsilon }

\inferrule*[Right=E-Proj]
  { \mathcal{G} \vdash \shat{e} : \mathcal{B} \elmark e \\
    \kw{lookup}(\mathcal{B}, n) = \mathcal{A} }
  { \mathcal{G} \vdash \shat{e}.n : \mathcal{A} \elmark e.n }

\inferrule*[Right=E-Lam]
  { \mathcal{G} \tand \mathcal{A} \vdash \shat{e} : \mathcal{B} \elmark e }
  { \mathcal{G} \vdash \slam\,\mathcal{A}.\,\shat{e} : \mathcal{A} \tarr \mathcal{B} \elmark \lambda\lsem\mathcal{A}\rsem.\,e }

\inferrule*[Right=E-Box]
  { \mathcal{G} \vdash \shat{e}_1 : \mathcal{G}' \elmark e_1 \\
    \mathcal{G}' \vdash \shat{e}_2 : \mathcal{A} \elmark e_2 }
  { \mathcal{G} \vdash \shat{e}_1 \sbox \shat{e}_2 : \mathcal{A} \elmark e_1 \triangleright e_2 }

\inferrule*[Right=E-Clos]
  { \mathcal{G} \vdash \shat{e}_1 : \mathcal{G}' \elmark e_1 \\
    \mathcal{G}' \tand \mathcal{A} \vdash \shat{e}_2 : \mathcal{B} \elmark e_2 }
  { \mathcal{G} \vdash \sclos{\shat{e}_1}{\mathcal{A}}{\shat{e}_2} : \mathcal{A} \tarr \mathcal{B} \elmark \cclos{e_1}{\lsem\mathcal{A}\rsem}{e_2} }

\inferrule*[Right=E-App]
  { \mathcal{G} \vdash \shat{e}_1 : \mathcal{A} \tarr \mathcal{B} \elmark e_1 \\
    \mathcal{G} \vdash \shat{e}_2 : \mathcal{A} \elmark e_2 }
  { \mathcal{G} \vdash \shat{e}_1\;\shat{e}_2 : \mathcal{B} \elmark e_1\;e_2 }

\inferrule*[Right=E-DMrg]
  { \mathcal{G} \vdash \shat{e}_1 : \mathcal{A} \elmark e_1 \\
    \mathcal{G} \tand \mathcal{A} \vdash \shat{e}_2 : \mathcal{B} \elmark e_2 }
  { \mathcal{G} \vdash \shat{e}_1 \sdmrg \shat{e}_2 : \mathcal{A} \tand \mathcal{B} \elmark e_1 \fatcomma e_2 }

\inferrule*[Right=E-NMrg]
  { \mathcal{G} \vdash \shat{e}_1 : \mathcal{A} \elmark e_1 \\
    \mathcal{G} \vdash \shat{e}_2 : \mathcal{B} \elmark e_2 }
  { \mathcal{G} \vdash \shat{e}_1 \snmrg \shat{e}_2 : \mathcal{A} \tand \mathcal{B} \elmark
    (\lambda\lsem\mathcal{G}\rsem.\,(\query.0 \triangleright e_1) \fatcomma (\query.1 \triangleright e_2))\;\query }

\inferrule*[Right=E-Rcd]
  { \mathcal{G} \vdash \shat{e} : \mathcal{A} \elmark e }
  { \mathcal{G} \vdash \srec{l}{\shat{e}} : \{l:\mathcal{A}\} \elmark \{l=e\} }

\inferrule*[Right=E-RProj]
  { \mathcal{G} \vdash \shat{e} : \mathcal{B} \elmark e \\
    l : \mathcal{A} \in \mathcal{B} }
  { \mathcal{G} \vdash \shat{e}.l : \mathcal{A} \elmark e.l }

\inferrule*[Right=E-Let]
  { \mathcal{G} \vdash \shat{e}_1 : \mathcal{A} \elmark e_1 \\
    \mathcal{G} \tand \mathcal{A} \vdash \shat{e}_2 : \mathcal{B} \elmark e_2 }
  { \mathcal{G} \vdash \slet{\shat{e}_1}{\mathcal{A}}{\shat{e}_2} : \mathcal{B} \elmark (\lambda\lsem\mathcal{A}\rsem.\,e_2)\;e_1 }

\inferrule*[Right=E-Open]
  { \mathcal{G} \vdash \shat{e}_1 : \{l:\mathcal{A}\} \elmark e_1 \\
    \mathcal{G} \tand \mathcal{A} \vdash \shat{e}_2 : \mathcal{B} \elmark e_2 }
  { \mathcal{G} \vdash \sopen{\shat{e}_1}{\shat{e}_2} : \mathcal{B} \elmark (\lambda\lsem\mathcal{A}\rsem.\,e_2)\;(e_1.l) }
\end{mathpar}

\subsubsection*{Module Rules}

Helper for functor result types:
\[
  \kw{mkMT}(\mathcal{A}, \mathcal{B}) =
  \begin{cases}
    \mathcal{A} \tarr_m \mathcal{M} & \text{if } \mathcal{B} = \kw{Mod}\;\mathcal{M} \\
    \mathcal{A} \tarr_m \kw{Intf}\;\mathcal{B} & \text{otherwise}
  \end{cases}
\]

\begin{mathpar}
\inferrule*[Right=E-Struct-S]
  { \varepsilon \vdash \shat{e} : \mathcal{B} \elmark e }
  { \mathcal{G} \vdash \Struct{\smark}{\shat{e}} : \kw{Mod}\;(\kw{Intf}\;\mathcal{B}) \elmark \varepsilon \triangleright e }

\inferrule*[Right=E-Struct-O]
  { \mathcal{G} \vdash \shat{e} : \mathcal{B} \elmark e }
  { \mathcal{G} \vdash \Struct{\omark}{\shat{e}} : \kw{Mod}\;(\kw{Intf}\;\mathcal{B}) \elmark \query \triangleright e }

\inferrule*[Right=E-Functor-S]
  { \varepsilon \tand \mathcal{A} \vdash \shat{e} : \mathcal{B} \elmark e \\
    \mathcal{M} = \kw{mkMT}(\mathcal{A}, \mathcal{B}) }
  { \mathcal{G} \vdash \Functor{\smark}{\mathcal{A}}{\shat{e}} : \kw{Mod}\;\mathcal{M} \elmark \varepsilon \triangleright \lambda\lsem\mathcal{A}\rsem.\,e }

\inferrule*[Right=E-Functor-O]
  { \mathcal{G} \tand \mathcal{A} \vdash \shat{e} : \mathcal{B} \elmark e \\
    \mathcal{M} = \kw{mkMT}(\mathcal{A}, \mathcal{B}) }
  { \mathcal{G} \vdash \Functor{\omark}{\mathcal{A}}{\shat{e}} : \kw{Mod}\;\mathcal{M} \elmark \lambda\lsem\mathcal{A}\rsem.\,e }

\inferrule*[Right=E-MClos]
  { \mathcal{G} \vdash \shat{e}_1 : \mathcal{G}' \elmark e_1 \\
    \mathcal{G}' \tand \mathcal{A} \vdash \shat{e}_2 : \mathcal{B} \elmark e_2 \\
    \mathcal{M} = \kw{mkMT}(\mathcal{A}, \mathcal{B}) }
  { \mathcal{G} \vdash \mclos{\shat{e}_1}{\mathcal{A}}{\shat{e}_2} : \kw{Mod}\;\mathcal{M} \elmark \cclos{e_1}{\lsem\mathcal{A}\rsem}{e_2} }

\inferrule*[Right=E-Link]
  { \mathcal{G} \vdash \shat{e}_1 : \kw{Mod}\;(\mathcal{A} \tarr_m \mathcal{M}) \elmark e_1 \\
    \mathcal{G} \vdash \shat{e}_2 : \mathcal{A} \elmark e_2 }
  { \mathcal{G} \vdash \Link{\shat{e}_1}{\shat{e}_2} : \kw{Mod}\;\mathcal{M} \elmark e_1\;e_2 }
\end{mathpar}

\section{Metatheory}
%%% ============================================================

All theorems are mechanized in Coq and validated via property-based
testing in Haskell (500 tests each, all passing).

\begin{theorem}[Type-Preserving Elaboration]
If\; $\mathcal{G} \vdash \shat{e} : \mathcal{A} \elmark e$,
then $\lsem \mathcal{G} \rsem \vdash e : \lsem \mathcal{A} \rsem$.
\end{theorem}

\begin{theorem}[Uniqueness of Elaboration]
If\; $\mathcal{G} \vdash \shat{e} : \mathcal{A}_1 \elmark e_1$
and  $\mathcal{G} \vdash \shat{e} : \mathcal{A}_2 \elmark e_2$,
then $\mathcal{A}_1 = \mathcal{A}_2$ and $e_1 = e_2$.
\end{theorem}

\begin{theorem}[Semantics Preservation]
If\; $\mathcal{G} \vdash \shat{e} : \mathcal{A} \elmark e$,
$\sval\;\hat{\rho}$,
$\hat{\rho} \vdash \shat{e} \Rightarrow \shat{v}$,
and $\hat{\rho} \sim \rho$ at type $\mathcal{G}$,
then there exists $v$ such that
$\rho \vdash e \Rightarrow v$
and $\shat{v} \sim v$ at type $\mathcal{A}$.
\end{theorem}

\begin{theorem}[Separate Compilation]
If\; $\mathcal{G} \vdash \shat{e}_1 : \kw{Mod}\;(\mathcal{A} \tarr_m \mathcal{M}) \elmark e_1$
and  $\mathcal{G} \vdash \shat{e}_2 : \mathcal{A} \elmark e_2$,
then $\mathcal{G} \vdash \Link{\shat{e}_1}{\shat{e}_2} : \kw{Mod}\;\mathcal{M} \elmark e_1\;e_2$.
\end{theorem}

\begin{theorem}[Sandbox Struct Isolation]
If\; $\sval\;\hat{\rho}_1$, $\sval\;\hat{\rho}_2$,
$\hat{\rho}_1 \vdash \Struct{\smark}{\shat{e}} \Rightarrow \shat{v}_1$,
and  $\hat{\rho}_2 \vdash \Struct{\smark}{\shat{e}} \Rightarrow \shat{v}_2$,
then $\shat{v}_1 = \shat{v}_2$.
\end{theorem}

\begin{theorem}[Sandbox Functor Isolation]
If\; $\sval\;\hat{\rho}_1$ and $\sval\;\hat{\rho}_2$,
then
$\hat{\rho}_1 \vdash \Functor{\smark}{\mathcal{A}}{\shat{e}} \Rightarrow \mclos{\hat{\varepsilon}}{\mathcal{A}}{\shat{e}}$
and
$\hat{\rho}_2 \vdash \Functor{\smark}{\mathcal{A}}{\shat{e}} \Rightarrow \mclos{\hat{\varepsilon}}{\mathcal{A}}{\shat{e}}$.
\end{theorem}

\begin{theorem}[Secure Linking]
If a sandboxed functor's body is closed (does not depend on the argument),
then the result of\/ $\kw{Link}$ is independent of both the linking
environment and the argument value.
\end{theorem}

\begin{theorem}[NMrg Independence]
If\; $\mathcal{G} \vdash \shat{e}_1 \snmrg \shat{e}_2 : \mathcal{A} \tand \mathcal{B} \elmark e$
and  $\mathcal{G} \vdash \shat{e}_2 : \mathcal{B}' \elmark e_2$,
then $\mathcal{B} = \mathcal{B}'$.
\end{theorem}

\begin{theorem}[Full Pipeline]
If\; $\varepsilon \vdash \shat{e} : \mathcal{A} \elmark e$,
then $\lsem\varepsilon\rsem \vdash e : \lsem\mathcal{A}\rsem$
and there exists a value $v$ such that $\varepsilon \vdash e \hookrightarrow^* v$.
\end{theorem}
